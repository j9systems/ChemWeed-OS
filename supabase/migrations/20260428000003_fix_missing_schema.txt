-- sort_order on service_types
ALTER TABLE public.service_types
  ADD COLUMN IF NOT EXISTS sort_order integer;

-- per_visit pricing model
ALTER TABLE public.service_types
  DROP CONSTRAINT IF EXISTS service_types_pricing_model_check;
ALTER TABLE public.service_types
  ADD CONSTRAINT service_types_pricing_model_check
  CHECK (pricing_model IN ('per_acre', 'per_hour', 'flat_rate', 'per_visit'));

-- recommendation_notes on service_agreements
ALTER TABLE public.service_agreements
  ADD COLUMN IF NOT EXISTS recommendation_notes text;

UPDATE public.service_types SET sort_order = 2 WHERE name ILIKE '%KICK START%';
UPDATE public.service_types SET sort_order = 3 WHERE name ILIKE '%QUALITY ASSURANCE%';
UPDATE public.service_types SET sort_order = 4 WHERE name ILIKE '%QUALITY PERFECTION%';
