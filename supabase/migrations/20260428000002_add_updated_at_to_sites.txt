-- Fix: A trigger on the sites table references NEW.updated_at, but the column
-- doesn't exist, causing "record 'new' has no field 'updated_at'" on any update.

ALTER TABLE public.sites
  ADD COLUMN IF NOT EXISTS updated_at timestamptz NOT NULL DEFAULT now();

UPDATE public.sites SET updated_at = created_at WHERE updated_at = now();

-- Recreate trigger (idempotent)
DROP TRIGGER IF EXISTS set_sites_updated_at ON public.sites;
CREATE TRIGGER set_sites_updated_at
  BEFORE UPDATE ON public.sites
  FOR EACH ROW
  EXECUTE FUNCTION public.handle_updated_at();
